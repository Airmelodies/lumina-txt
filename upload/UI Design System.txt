### **Section B: React + Tailwind Implementation**

**Design Notes:**
- Hardware-terminal aesthetic preserved through token usage
- IBM Plex fonts loaded via Google Fonts (fallbacks included)
- Scanline texture moved to optional utility class (performance-conscious)
- CSS-only hover/focus states, no JS for visual feedback
- Semantic HTML: `<header>`, `<aside>`, `<main>`, `<article>`
- Mobile-responsive: panels stack on small viewports

// Reusable TerminalCard component
export function TerminalCard({ children, className = '', active = false }) {
  return (
    <article
      className={`
        bg-[#111118] border border-[#22222f] rounded-[3px] p-3
        transition-colors duration-100
        ${active ? 'border-l-2 border-l-[#c8f060]' : ''}
        ${className}
      `}
    >
      {children}
    </article>
  );
}

// Reusable TerminalButton component
export function TerminalButton({ 
  children, 
  variant = 'default', 
  className = '', 
  ...props 
}) {
  const baseStyles = `
    font-mono text-[10px] tracking-[0.04em] uppercase
    px-3 py-1.5 rounded-[2px] transition-all duration-120
    focus:outline-none focus:ring-1 focus:ring-[#2e2e3e]
  `;
  
  const variants = {
    default: `
      bg-transparent border border-[#22222f] 
      text-[#55556a] hover:border-[#2e2e3e] hover:text-[#7a7a96]
    `,
    primary: `
      border border-[#7a9238] bg-[rgba(200,240,96,0.08)] 
      text-[#c8f060] hover:bg-[rgba(200,240,96,0.15)] 
      hover:border-[#c8f060] hover:shadow-[0_0_8px_rgba(200,240,96,0.15)]
    `,
    ghost: `
      bg-transparent border-none text-[#3a3a50] 
      hover:text-[#7a7a96] hover:bg-[#16161f]
    `,
  };

  return (
    <button 
      className={`${baseStyles} ${variants[variant]} ${className}`}
      {...props}
    >
      {children}
    </button>
  );
}
